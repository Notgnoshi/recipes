\begin{recipe}{Stuffed Hamburger Roll}{serves 4}{350\0}
    \ingredient[1]{loaf}{French bread}
    Cut bread lengthwise. Scoop out inside of top and bottom pieces. Break filling into small pieces.

    \ingredient[1]{lb}{Ground beef}
    Brown and drain ground beef.

    \ingredient[2]{cans}{Cream of chicken/mushroom soup}
    \ingredient[1]{cup}{Cubed Velveeta cheese}
    Mix soup, cheese, and bread pieces into the beef.

    \newstep
    Place filling into bread, and wrap loaf in tin foil. Bake at 350\0 for 35 minutes.
\end{recipe}

\begin{recipe}{Sausage Bread}{serves 4}{350\0}
    \ingredient[2]{loaves}{Frozen white bread dough}
    Cover dough and let rise in warm place until doubled in size. Flatten each loaf into a $16 \times 12$ inch rectangle.

    \ingredient[1]{lb}{Ground sausage}
    Brown and drain sausage. Preheat oven to 350\0.

    \ingredient[3]{}{Eggs}
    \ingredient[$2\frac{1}{2}$]{cups}{Shredded mozarella cheese}
    \ingredient[1]{tsp}{Dried basil}
    \ingredient[1]{tsp}{Parsley flakes}
    \ingredient[1]{tsp}{Crushed rosemary}
    \ingredient[1]{tsp}{Garlic powder}
    Stir in two eggs, cheese, and seasonings. Spread mixture over both dough rectangles to within 1 inch of the edges.

    \newstep
    Start with short side, and roll up rectangles like a jelly roll. Pinch seams to seal and place on greased baking sheet.

    \newstep
    Whisk remaining egg and brush over tops of loaves. Bake at 350\0 for 25--30 minutes, or until golden brown.
\end{recipe}
